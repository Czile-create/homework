有n个节点的无向图，用邻接矩阵储存，回答下列问题

图中的边数

size_t getSizeOfEdge(vector<vector<size_t>> &p) {
    return accumulate (
        p.begin(), 
        p.end(),
        0, 
        [](size_t init, auto &t) {
            return init + accumulate(
                t.begin(),
                t.end(),
                0, 
                [](size_t init, auto &e) {
                    return init + bool(e);
                }
            );
        }
    );
}

任意两个顶点是否有边相连

bool islinked(
    vector<vector<size_t>> &p,
    size_t i,
    size_t j
) const noexcept {
    return p[i][j];
}

任意一个顶点的度

size_t numOfEdge(
    vector<vector<size_t>> &p,
    size_t i
) const noexcept {
    return accumulate(
        p[i].begin(),
        p[i].end(),
        0, 
        [](size_t init, auto &t) {
            return init + bool(t);
        }
    );
}
